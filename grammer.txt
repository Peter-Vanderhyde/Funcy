                     | 'break' ';'
                     | 'continue' ';'

                     | 'while' Equality Scope

                     | 'else' Scope
                     | 'for' Statement ',' Relation ',' Statement Scope
                     | 'func' Identifier '(' [ Identifier [',' Identifier]* ]? ')' Scope

Statement  : [Identifier ('=' | '+=' | '-=' | '*=' | '/=')] LogicalOr ';'

LogicalOr  : LogicalAnd ['or' LogicalAnd]*

LogicalAnd : Equality ['and' Equality]*

Equality   : Relation [ ('==' | '!=') Relation ]

Relation   : Expression [ ( '<' | '<=' | '>' | '>=' ) Expression ]

Expression : Term [ ('+' | '-') Term ]*

Term       : Factor [ ('*' | '/' | '//' | '%') Factor ]*

Factor     : [ ('+' | '-') ]? Power

Power      : LogicalNot [ ('^' | '**') Factor ]?

LogicalNot : ['not' | '!']* Atom

Primary    : Collection | FunctionCall | ( '(' LogicalOr ')' ) | Scope

Collection : Atom | List

Atom       : Identifier | Number | Boolean | String

Number     : Integer | Float

FunctionCall : Identifier '(' [LogicalOr [',' LogicalOr]* ]? ')'

Identifier : (a-zA-Z_)[a-zA-Z_0-9]*

Boolean    : true | false

String     : '"' [^"]* '"'

List       : '[' [LogicalOr [',' LogicalOr]* ]? ']'

Float      : [0-9]+ '.' [0-9]+
