Expression : Term [ ('+' | '-') Term ]*

Term       : Factor [ ('*' | '/') Factor ]*

Factor     : [ ('+' | '-') ]? Power

Power      : Primary [ '^' Factor ]?

Primary    : Number | '(' Expression ')'

Number     : Integer | Float

Integer    : [0-9]+

Float      : [0-9]+ '.' [0-9]+


Statement : [Identifier '='] Expression

Identifier : (a-zA-Z_)[a-zA-Z_0-9]*
